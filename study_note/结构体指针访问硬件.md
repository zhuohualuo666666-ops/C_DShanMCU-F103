<!-- 结构体指针访问硬件 -->
struct GPIORegs {
    volatile unsigned int crl;    //偏移地址0x00(0)
    volatile unsigned int crh;    //偏移地址0x04(4)  
    volatile unsigned int idr;    //偏移地址0x08(8)
    volatile unsigned int odr;    //偏移地址0x0C(12)
    volatile unsigned int bsrr;    //偏移地址0x10(16)
    volatile unsigned int brr;    //偏移地址0x14(20)
    volatile unsigned int lckr;    //偏移地址0x18(24)
    volatile unsigned int afio;    //偏移地址0x1C(28)
};

使用这个结构体将不同作用的寄存器包含进去，
按照一定的顺序在结构体中排列，
相对于基地址有不同的偏移地址

struct GPIORegs f;//定义结构体变量
用结构体变量无法操作GPIO寄存器，例：
f.odr=1;
这里只是在定义结构体变量f开辟的内存中将odr这个变量的值修改为1，
并没有操作到寄存器，
因此使用结构体指针:

struct GPIORegs *p;
p=(struct GPIORegs *)0x40011000;
或
struct GPIORegs *p = (struct GPIORegs *)0x40011000;
这两种写法等效


p->odr &= ~(1<<13);  /* 点亮LED1 */   
p->odr |= (1<<13);  /* 熄灭LED1 */
（1）读
（2）位与，位或
（3）写
