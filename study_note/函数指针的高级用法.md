<!-- 函数指针的高级用法 -->

int add(int a,int b)
{
    return a+b;
}

int sub(int a,int b)
{
    return a-b;
}

int multi(int a,int b)
{
    return a*b;
}

int div(int a,int b)
{
    return a/b;
}

typedef int (*f_type)(int a,int b);



1.函数指针参数
int process(f_type f,int a,int b)//process是一个函数，参数有函数指针f，类型为f_type，即int(*)(int,int)；变量a,b
{
    int result=f(a,b);//跳到f所表示的地址执行
    return result;
}

int mymain(void)
{
    int v1=4,v2=2;
    int result;

    result=process(add,v1,v2);//执行add函数
    printf("result 1=%d\r\n",result);
    result=process(sub,v1,v2);//执行sub函数
    printf("result 2=%d\r\n",result);
    result=process(multi,v1,v2);//执行multi函数
    printf("result 3=%d\r\n",result);
    result=process(div,v1v2);//执行啊div函数
    printf("result 4=%d\r\n",result);
    return 0;
}

调用process函数时，开辟4*3=12个字节的空间（地址4字节，两个int类型变量8字节）；
process(add,v1,v2)这一句将add函数地址传给作为参数的函数指针f，v1,v2传给变量a,b,
在process函数中，将process的a,b传递给add函数，参数共传递2次，进行3层调用。

2.函数指针数组
int process(f_type f,int a,int b)//process是一个函数，参数有函数指针f，变量a,b
{
    int result=f(a,b);//跳到f所表示的地址执行
    return result;
}

int mymain(void)
{
    int v1=4,v2=2;
    int result;

    result=process(add,v1,v2);//执行add函数
    printf("result 1=%d\r\n",result);
    result=process(sub,v1,v2);//执行sub函数
    printf("result 2=%d\r\n",result);
    result=process(multi,v1,v2);//执行multi函数
    printf("result 3=%d\r\n",result);
    result=process(div,v1v2);//执行啊div函数
    printf("result 4=%d\r\n",result);

    f_type f_array[4]={add,sub,multi,div};//数组f_array的每个成员都是一个f_type类型的指针,存放函数的地址
    for(int i=0;i<4:i++)
    {
        result=f_array[i](v1,v2);
        printf("f_array[%d]'s result=%d\r\n",i,result);
    }
    return 0;
}


f_type f_array[4]={add,sub,multi,div}这一句开辟了4*4=16个字节的空间，存放add,sub,multi,div四个函数的地址
result=f_array[i](v1,v2)这一句用f_array[i]的值作为地址执行对应的函数，i为0执行add,1执行sub

for循环里实现了将main函数里定义的变量分别进行加、减、乘、除运算

3.结构体中的函数指针
struct person{
    char *name;
    int sex;
    void (*printf)(struct person *p);//printf这个函数的函数指针，通过person类型结构体的指针访问结构体中的内容
}//开辟12个字节空间

int process(f_type f,int a,int b)//process是一个函数，参数有函数指针f，变量a,b
{
    int result=f(a,b);//跳到f所表示的地址执行
    return result;
}

void printName(struct person *p)
{
    printf("Name:%s\r\n",p->name);
}

void printAll(struct person *p)
{
    printf("Name:%s\r\n",p->name);
    printf("Sex:%s\r\n",p->sex ? "Male":"Female");
}


int mymain(void)
{
    int v1=4,v2=2;
    int result;

    result=process(add,v1,v2);//执行add函数
    printf("result 1=%d\r\n",result);
    result=process(sub,v1,v2);//执行sub函数
    printf("result 2=%d\r\n",result);
    result=process(multi,v1,v2);//执行multi函数
    printf("result 3=%d\r\n",result);
    result=process(div,v1,v2);//执行啊div函数
    printf("result 4=%d\r\n",result);

    f_type f_array[4]={add,sub,multi,div};//数组f_array的每个成员都是一个f_type类型的指针,存放函数的地址
    for(int i=0;i<4;i++)
    {
        result=f_array[i](v1,v2);
        printf("f_array[%d]'s result=%d\r\n",i,result);
    }

    struct person per1={"zhangsan",1,printName};
    struct person per2={"lili",0,printAll};
    
    per1.print(&per1);
    per2.print(&per2);

    return 0;
}


struct person per={"zhangsan",1,printName}在内存中开辟12个字节空间，存放zhangsan这个字符串的首地址，性别1，printName函数的地址
struct person per={"lili",0,printAll}在内存中开辟12个字节空间，存放lili这个字符串的首地址，性别1，printAll函数的地址
per1.print(&per1)调用printName函数，传入per1这个结构体的指针，打印出姓名
per2.print(&per2)调用printAll函数，传入per2这个结构体的指针，打印出姓名性别
