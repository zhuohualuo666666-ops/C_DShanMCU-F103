<!-- 函数指针的基础使用 -->

int add(int a,int b)
{
    return a+b;
}

int sub(int a,int b)
{
    return a-b;
}

int multi(int a,int b)
{
    return a*b;
}

int div(int a,int b)
{
    return a/b;
}


int mymain(void)
{
    int v1=4,v2=2;
    int result;
    char c;
    int (*f)(int a,int b);

    printf("enter + or - or * or / \r\n");
    scanf("%c",&c);

    if(c=='+')
    f=add;
    else if(c=='-')
    f=sub;
    else if(c=='*')
    f=multi;
    else if(c=='/')
    f=div;

    result=f(v1,v2);
    printf("result1 of v1,v2=%d\r\n",result);

    v1=6;
    v2=3;

    result=f(v1,v2);
    printf("result1 of v1,v2=%d\r\n",result);
    
    return 0;
}


函数指针本质上是一个变量，用来存储地址,
int (*f)(int a,int b)这一句执行完，f是函数指针，是一个变量；

函数在调用时其函数名代表的含义是它所在的空间的首地址，
f=add这一句在调用add的时候编译器自动将add转为&add；

int (*f)(int a,int b)执行完，变量f的类型名是 
int(*)(int,int),
int add(int a,int b)这个函数定义完，add的类型名是
int (int,int),
调用时编译器自动加上取址&，因为函数不像变量一样直接被传递或复制，几乎总以int(*)(int,int)出现，
因为类型与变量f相同，因此可以直接赋值，如果用其他类型变量赋add，可以强转add类型，因为是地址，但需要确保目标地址对应的函数的参数、返回值与指针类型相同，防止出现异常。

使用函数指针后，对于加减乘除的判断就不需要每次手写判断




typedef的作用：将变量变成类型

int a;//a是变量
int *a;//a是指针变量
typedef int a;//a是int类型
typedef int *a;//a是int *类型

int add(int a,int b);//add是函数
int (*add)(int a,int b);//add是函数指针，是一个变量
typedef int (*add)(int a,int b);//add是函数指针类型，是一个类型

 定义一个函数指针类型，
 类型名 int(*)(int,int),
 用f_type指代它
 */
 
 f_type f;  //定义int(*)(int,int)类型的函数指针f
